/**
 * 화면 아이디 기준의 서비스 호출 주소를 가져온다.<p>
 * @param restId {String} REST API ID
 */
mint_rest.prototype.getServiceUrl = function(restId) {
	try {
		var url = mint_rest.services[restId];
		if( !url ) {
		//	alert("서비스 아이디에 해당하는 호출 주소가 존재하지 않습니다. 확인하여 주십시오.");
			return null;
		}
		return mint.baseServiceUrl + url + "&isTest=" + mint.isTestMode + "";
	} catch( e ){
		mint.common.errorLog(e, {"fnName" : "mint.rest.getServiceUrl"});
	}
};
